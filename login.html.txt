<script>
// ─── FIX: APP_URL injetada diretamente pelo backend via scriptlet GAS ──────
// Isso elimina a corrida entre o google.script.run assíncrono e o clique do usuário.
var APP_URL = '<?= _APP_URL ?>';
if (!APP_URL) APP_URL = window.location.href.replace(/[?#].*$/, '');

var _erros = 0, _bloqAte = 0, _submetendo = false;

function tgPass() {
  var i = document.getElementById('f-p'), on = i.type === 'password';
  i.type = on ? 'text' : 'password';
  document.getElementById('ico-e').style.display = on ? 'none' : '';
  document.getElementById('ico-o').style.display = on ? '' : 'none';
}

function showErr(msg) {
  document.getElementById('err-txt').textContent = msg || 'Usuário ou senha incorretos.';
  document.getElementById('err-box').classList.add('show');
  ['f-u', 'f-p'].forEach(function(id) {
    var el = document.getElementById(id);
    el.classList.add('err-f', 'shake');
    el.addEventListener('animationend', function() { el.classList.remove('shake'); }, { once: true });
    el.addEventListener('input', function() {
      el.classList.remove('err-f');
      document.getElementById('err-box').classList.remove('show');
    }, { once: true });
  });
}

function setBloq(seg) {
  var btn = document.getElementById('btn-e'), txt = document.getElementById('btn-t');
  _bloqAte = Date.now() + seg * 1000;
  btn.disabled = true;
  var iv = setInterval(function() {
    var r = Math.ceil((_bloqAte - Date.now()) / 1000);
    if (r <= 0) { clearInterval(iv); btn.disabled = false; txt.textContent = 'Entrar no painel'; return; }
    txt.textContent = 'Aguarde ' + r + 's…';
  }, 500);
}

// ─── FIX 2: Navegação robusta via link DOM com target="_top" ────────────────
// No Google Apps Script o app roda dentro de um iframe.
// Só target="_top" garante que a navegação quebre o iframe e carregue no topo.
function navegarPara(urlDestino) {
  // Tenta primeiro via google.script.history (mantém SPA limpo)
  try {
    // Cria elemento <a> e adiciona ao DOM antes de clicar (necessário em alguns browsers)
    var a = document.createElement('a');
    a.href   = urlDestino;
    a.target = '_top';
    a.rel    = 'noopener';
    a.style.display = 'none';
    document.body.appendChild(a);
    a.click();
    // Remove o elemento após click para não poluir o DOM
    setTimeout(function() { try { document.body.removeChild(a); } catch(e){} }, 1000);
  } catch(e) {
    // Fallback: força navegação direta no contexto pai
    try { window.top.location.href = urlDestino; } catch(e2) {
      window.location.href = urlDestino;
    }
  }
}

function entrar() {
  if (Date.now() < _bloqAte) return;
  if (_submetendo) return;

  var u = document.getElementById('f-u').value.trim();
  var p = document.getElementById('f-p').value;
  document.getElementById('err-box').classList.remove('show');

  if (!u) { document.getElementById('f-u').focus(); showErr('Informe o usuário.'); return; }
  if (!p) { document.getElementById('f-p').focus(); showErr('Informe a senha.'); return; }

  var btn = document.getElementById('btn-e'),
      sp  = document.getElementById('sp'),
      txt = document.getElementById('btn-t');

  _submetendo = true;
  btn.disabled = true;
  sp.style.display = 'block';
  txt.textContent = 'Verificando…';

  google.script.run
    .withSuccessHandler(function(r) {
      if (r && r.ok) {
        // ─── FIX 3: Salva credenciais na sessão ANTES de navegar ────────────
        try {
          sessionStorage.setItem('samu_token',   r.token);
          sessionStorage.setItem('samu_usuario', r.nome);
          sessionStorage.setItem('samu_admin',   r.admin ? '1' : '0');
        } catch(e) {}

        txt.textContent = 'Bem-vindo, ' + r.nome.charAt(0) + r.nome.slice(1).toLowerCase() + '!';

        // ─── FIX 4: Monta URL com token e navega para o painel ──────────────
        // APP_URL já foi obtida do backend; se ainda estiver vazia usa fallback
        var base = APP_URL || window.location.href.replace(/[?#].*$/, '');
        var urlFinal = base + '?p=painel&t=' + encodeURIComponent(r.token);
        navegarPara(urlFinal);

      } else {
        sp.style.display = 'none';
        _submetendo = false;
        btn.disabled = false;
        txt.textContent = 'Entrar no painel';
        _erros++;
        showErr(r && r.erro ? r.erro : 'Usuário ou senha incorretos.');
        if (_erros >= 3) {
          var seg = Math.min(15 * (_erros - 2), 120);
          showErr('Muitas tentativas. Aguarde ' + seg + 's.');
          setBloq(seg);
        }
        document.getElementById('f-p').value = '';
        document.getElementById('f-p').focus();
      }
    })
    .withFailureHandler(function(e) {
      _submetendo = false;
      sp.style.display = 'none';
      btn.disabled = false;
      txt.textContent = 'Entrar no painel';
      showErr('Erro de conexão: ' + (e && e.message ? e.message : 'tente novamente.'));
    })
    .verificarLogin(u, p);
}

window.onload = function() {
  // Limpa qualquer sessão anterior ao exibir a tela de login
  try {
    sessionStorage.removeItem('samu_token');
    sessionStorage.removeItem('samu_usuario');
    sessionStorage.removeItem('samu_admin');
  } catch(e) {}
  document.getElementById('f-u').focus();
};
