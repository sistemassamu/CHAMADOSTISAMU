  function configurarDragDrop(id, cb) {
    var z = document.getElementById(id);
    if (!z) return;
    z.addEventListener('dragover', function(e){ e.preventDefault(); z.classList.add('drag-over'); });
    z.addEventListener('dragleave', function(e){ if(!z.contains(e.relatedTarget)) z.classList.remove('drag-over'); });

  function processarArquivos(files, tipo) {
    var erros = [];
    if (tipo === 'fotos') {
      files.forEach(function(f){
        if (fotosLista.length >= MAX_FOTOS){ erros.push('"'+f.name+'" ignorado: limite atingido.'); return; }
        if (f.size > MAX_BYTES){ erros.push('"'+f.name+'" excede 4 MB.'); return; }
        if (!f.type.startsWith('image/')){ erros.push('"'+f.name+'" não é imagem.'); return; }
        fotosLista.push(f);
      });
      renderizarLista('fotos');
    } else {
      files.forEach(function(f){
        var n=f.name.toLowerCase();
        if (!EXT_DOC.some(function(e){return n.endsWith(e);}) && TIPOS_DOC.indexOf(f.type)===-1){ erros.push('"'+f.name+'" tipo não permitido.'); return; }
        if (f.size > MAX_BYTES){ erros.push('"'+f.name+'" excede 4 MB.'); return; }
        if (docsLista.some(function(d){return d.name===f.name;})){ erros.push('"'+f.name+'" já adicionado.'); return; }
        docsLista.push(f);
      });
      renderizarLista('docs');
    }
    if (erros.length) mostrarErro(erros.join(' | '));
  }

      thumb.className = 'fi-icon-label ' + cls;
      thumb.textContent = lbl;

  _chatPollTimer = setInterval(function() {
    if (_chatIdAtual !== id) { 
      clearInterval(_chatPollTimer); 
      _chatPollTimer = null;
      return; 
    }
    carregarMsgsChat(id, true);
  }, 5000); // Reduzindo para 5 segundos para melhor responsividade
}

function carregarMsgsChat(id, chatAtivo) {

    .withFailureHandler(function(e) {
      console.error('Erro ao carregar mensagens:', e);
      console.log('Detalhes do erro:', e.message);
      try {
        document.getElementById('chat-msgs-' + id).innerHTML = '<div style="text-align:center;color:red;padding:10px">Erro ao carregar mensagens: ' + e.message + '</div>';
      } catch(err) {
        console.log('Erro ao mostrar erro no UI:', err);
      }
    })
    .buscarMensagens(id, false, ''); // Token vazio para usuários
}

  function renderMsgsChat(id, msgs, ativo) {
    var el = document.getElementById('chat-msgs-' + id);
    if (!el) return;
    if (!msgs || msgs.length === 0) { el.innerHTML = ''; return; }
    el.innerHTML = msgs.map(function(m) {
      var lado = m.de === 'USUARIO' ? 'usuario' : 'ti';
      var quem = m.de === 'TI' ? (m.nome || 'TI') : 'Você';
      return '<div class="chat-bubble ' + lado + '">'
        + escHtml(m.msg)
        + '<div class="chat-bubble-meta">' + escHtml(quem) + ' · ' + escHtml(m.hora) + '</div>'
        + '</div>';
    }).join('');
    el.scrollTop = el.scrollHeight;
  }

function enviarChatUsuario(id, nomeEnc) {
  var txt = document.getElementById('chat-txt-' + id);
  var btn = document.getElementById('chat-btn-' + id);
  if (!txt || !txt.value.trim()) return;
  var msg = txt.value.trim();
  var nome = decodeURIComponent(nomeEnc);
  txt.disabled = true; btn.disabled = true;
  
  google.script.run
    .withSuccessHandler(function(r) {
      txt.disabled = false; btn.disabled = false;
      if (r && r.ok) {
        txt.value = '';
        carregarMsgsChat(id, true); // Forçar reload imediato após envio
      } else {
        mostrarErro(r && r.erro ? r.erro : 'Erro ao enviar.');
      }
    })
    .withFailureHandler(function(e) {
      txt.disabled = false; btn.disabled = false;
      mostrarErro('Erro ao enviar: ' + e.message);
    })
    .enviarMensagemChat({ id: id, de: 'USUARIO', nome: nome, msg: msg }); 
}
