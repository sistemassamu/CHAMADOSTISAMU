function navegarPara(urlDestino) {
  try {
    var a = document.createElement('a');
    a.target = '_top';
    a.href = urlDestino;
    document.body.appendChild(a); 
    a.click();
  } catch(e) {
    try {
      window.top.location.href = urlDestino;
    } catch(err) {
      window.open(urlDestino, '_top');
    }
  }
}

function renderizarIndicadores(d) {
  var indicadores = [
    { id:'setores',    titulo:'Setor solicitante',          dados:d.setores,    cor:'#F05252', full:true  },
    { id:'naturezas',  titulo:'Natureza da atividade',      dados:d.naturezas,  cor:'#60A5FA', full:false },
    { id:'prioridades',titulo:'Classificação de prioridade',dados:d.prioridades,cor:'#FBBF24', full:false },
    { id:'turnos',     titulo:'Turno das atividades',       dados:d.turnos,     cor:'#FBBF24', full:false },
    { id:'dias',       titulo:'Dias da semana',             dados:d.dias,       cor:'#2DD4BF', full:false }
  ];

  var grid = document.getElementById('grid');
  grid.innerHTML = '';

  indicadores.forEach(function(ind) {
    var labels  = Object.keys(ind.dados);
    var valores = Object.values(ind.dados);
    var total   = valores.reduce(function(a,b){return a+b;}, 0);
    var paleta  = PALETAS[ind.id] || ['#E53935'];
    var destCor = DESTAQUE[ind.id] || paleta[0];

    if (total === 0) return;

    var card = document.createElement('div');
    card.className = 'ind-card' + (ind.full ? ' full' : '');
    card.innerHTML =
      '<div class="ind-header">'
      + '<div class="ind-title"><div class="ind-dot" style="background:'+ind.cor+'"></div>'+esc(ind.titulo)+'</div>'
      + '<div class="ind-count">'+total+' registros</div>'
      + '</div>'
      + '<div class="ind-body">'
      +   '<div class="ind-bars" id="bars-'+ind.id+'"></div>'
      +   '<div class="ind-pie-wrap">'
      +     '<div class="pie-canvas-wrap"><canvas id="pie-'+ind.id+'"></canvas></div>'
      +     '<div class="pie-legend" id="leg-'+ind.id+'"></div>'
      +   '</div>'
      + '</div>';
    grid.appendChild(card);

    var barsEl = document.getElementById('bars-'+ind.id);
    var max    = Math.max.apply(null, valores) || 1;
    labels.forEach(function(lbl, i) {
      var v      = valores[i];
      var pct    = (v / total * 100).toFixed(1);
      var isTop  = i === 0; 
      var corBarra = isTop ? destCor : paleta[i % paleta.length];
      var trackOp  = isTop ? 'rgba(255,255,255,.10)' : 'rgba(255,255,255,.05)';
      var row = document.createElement('div');
      row.className = 'bar-row' + (isTop ? ' destaque' : '');
      row.innerHTML =
        '<div class="bar-label" title="'+esc(lbl)+'">'+esc(lbl)+'</div>'
        + '<div class="bar-track" style="background:'+trackOp+'">'
        +   '<div class="bar-fill" style="width:0%;background:'+corBarra+'" data-w="'+(v/max*100).toFixed(1)+'%"></div>'
        + '</div>'
        + '<div class="bar-pct">'+pct+'%</div>'
        + '<div class="bar-num">'+v+'</div>';
      barsEl.appendChild(row);
    });

    setTimeout(function() {
      barsEl.querySelectorAll('.bar-fill').forEach(function(el) {
        el.style.width = el.getAttribute('data-w');
      });
    }, 80);

    var bgColors = labels.map(function(_, i) {
      return i === 0 ? destCor : paleta[i % paleta.length];
    });
    var ctx = document.getElementById('pie-'+ind.id).getContext('2d');
    charts[ind.id] = new Chart(ctx, {
      type: 'doughnut',
      data: {
        labels: labels,
        datasets: [{
          data: valores,
          backgroundColor: bgColors,
          borderColor: '#172A44',
          borderWidth: 2,
          hoverOffset: 6
        }]
      },
      options: {
        responsive:true, maintainAspectRatio:true, cutout:'62%',
        plugins: {
          legend: { display: false },
          tooltip: {
            backgroundColor:'#0C1729', borderColor:'rgba(255,255,255,.14)', borderWidth:1,
            titleColor:'#EEF3FF', bodyColor:'#90A8C4',
            callbacks: { label: function(c){ return ' '+c.parsed+' ('+((c.parsed/total)*100).toFixed(1)+'%)'; } }
          }
        },
        animation:{ animateRotate:true, duration:700 }
      }
    });

    var legEl = document.getElementById('leg-'+ind.id);
    var maxL  = Math.min(labels.length, 5);
    for (var i = 0; i < maxL; i++) {
      var pct = (valores[i] / total * 100).toFixed(1);
      var li  = document.createElement('div');
      li.className = 'pie-leg-item';
      li.innerHTML =
        '<div class="pie-leg-dot" style="background:'+bgColors[i]+'"></div>'
        + '<div class="pie-leg-label" title="'+esc(labels[i])+'">'+esc(labels[i])+'</div>'
        + '<div class="pie-leg-pct">'+pct+'%</div>';
      legEl.appendChild(li);
    }
    if (labels.length > 5) {
      var r = document.createElement('div');
      r.className = 'pie-leg-item';
      r.style.color = 'var(--text-3)';
      r.textContent = '+ '+(labels.length-5)+' outros';
      legEl.appendChild(r);
    }
  });

  grid.classList.add('visible');
}

window.onload = function() {
  var m = window.location.search.match(/[?&]t=([^&]+)/);
  if (m) {
    _tokenSessao = decodeURIComponent(m[1]);
    try { sessionStorage.setItem('samu_token', _tokenSessao); } catch(e){}
    try { google.script.history.replace(null, null, '?p=relatorio'); } catch(err){}
  } else {
    try { _tokenSessao = sessionStorage.getItem('samu_token') || ''; } catch(e){}
  }

  if (!_tokenSessao) {
    irParaLogin();
    return;
  }

  google.script.run
    .withSuccessHandler(function(r) {
      if (!r || !r.ok) {
        document.getElementById('loading-msg').textContent = 'Sessão inválida. Redirecionando...';
        setTimeout(irParaLogin, 1500);
        return;
      }
      carregar();
    })
    .withFailureHandler(function() {
      document.getElementById('loading-msg').textContent = 'Erro de conexão. Tente novamente.';
      setTimeout(irParaLogin, 1500);
    })
    .verificarTokenAcesso(_tokenSessao);
};

function exportarSlides() {
  document.getElementById('slides-loading').style.display = 'block';
  document.getElementById('slides-sucesso').style.display = 'none';
  document.getElementById('slides-erro').style.display   = 'none';
  document.getElementById('slides-backdrop').classList.add('open');
  document.getElementById('btn-slides').classList.add('loading');

  var mes = document.getElementById('sel-mes').value;
  var ano = document.getElementById('sel-ano').value;

  google.script.run
    .withSuccessHandler(function(res) {
      document.getElementById('btn-slides').classList.remove('loading');
      document.getElementById('slides-loading').style.display = 'none';
      if (res.ok) {
        document.getElementById('slides-url').href = res.url;
        document.getElementById('slides-sucesso').style.display = 'block';
      } else {
        document.getElementById('slides-erro-msg').textContent = res.erro || 'Erro desconhecido.';
        document.getElementById('slides-erro').style.display   = 'block';
      }
    })
    .withFailureHandler(function(e) {
      document.getElementById('btn-slides').classList.remove('loading');
      document.getElementById('slides-loading').style.display = 'none';
      document.getElementById('slides-erro-msg').textContent  = e.message || 'Falha na comunicação.';
      document.getElementById('slides-erro').style.display    = 'block';
    })
    .exportarParaSlides({ mes: mes, ano: ano }, _tokenSessao); 
}
