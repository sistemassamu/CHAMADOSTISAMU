      var a = document.createElement('a');

function carregarPchat(id, marcarLida) {
  console.log('carregarPchat chamado - ID:', id, 'Marcar lida:', marcarLida);
  
  google.script.run
    .withSuccessHandler(function(r) {
      console.log('Resposta carregarPchat:', r);
      if (!r) {
        console.log('Resposta vazia de carregarPchat');
        return;
      }
      
      if (r.erro) {
        console.log('Erro na resposta:', r.erro);
        return;
      }
      
      renderPchat(r.msgs || []);
      
      // Atualizar badge de mensagens não lidas
      if (r.msgs && Array.isArray(r.msgs)) {
        var nL = r.msgs.filter(function(m) {
          return m.de === 'USUARIO' && !m.lida;
        }).length;
        
        var badge = document.getElementById('tab-chat-badge');
        if (badge) {
          badge.textContent = nL > 0 ? nL : '';
          badge.style.display = nL > 0 ? 'inline' : 'none';
        }
      }
    })
    .withFailureHandler(function(e) {
      console.error('Erro ao carregar chat:', e);
      console.log('Detalhes do erro:', e.message);
    })
    .buscarMensagens(id, !!marcarLida, _tokenSessao);
}

function iniciarPollPchat(id){if(_pchatPollTimer)clearInterval(_pchatPollTimer);_pchatIdAtual=id;_pchatPollTimer=setInterval(function(){if(_pchatIdAtual!==id){clearInterval(_pchatPollTimer);return;}carregarPchat(id,false);},10000);}
function renderPchat(msgs){
  var el=document.getElementById('pchat-msgs');if(!el)return;
  if(!msgs||!msgs.length){el.innerHTML='';return;}
  el.innerHTML=msgs.map(function(m){
    var lado=m.de==='TI'?'ti':'usuario';var quem=m.de==='TI'?(m.nome||'TI'):m.nome||'Usuário';
    return '<div class="pchat-bubble '+lado+'">'+escH(m.msg)+'<div class="pchat-meta">'+escH(quem)+' · '+escH(m.hora)+(m.de==='USUARIO'&&!m.lida?' · <span style="color:var(--red)">não lida</span>':'')+'</div></div>';
  }).join('');
  el.scrollTop=el.scrollHeight;
}

function enviarChatTI() {
  var id = document.getElementById('r-id').value;
  var tec = document.getElementById('r-tecnico').value || 'TI';
  var txt = document.getElementById('pchat-txt');
  var btn = document.getElementById('pchat-btn');
  
  if (!txt || !txt.value.trim()) return;
  
  var msg = txt.value.trim();
  console.log('Enviando mensagem - ID:', id, 'Técnico:', tec, 'Mensagem:', msg);
  
  txt.disabled = true;
  btn.disabled = true;
  
  google.script.run
    .withSuccessHandler(function(r) {
      txt.disabled = false;
      btn.disabled = false;
      
      if (r && r.ok) {
        console.log('Mensagem enviada com sucesso');
        txt.value = '';
        carregarPchat(id, false); // Recarregar imediatamente
        
        // Mostrar feedback visual
        var tempDiv = document.createElement('div');
        tempDiv.style.cssText = 'position:fixed;top:10px;right:10px;background:green;color:white;padding:10px;border-radius:5px;z-index:9999;';
        tempDiv.textContent = 'Mensagem enviada!';
        document.body.appendChild(tempDiv);
        setTimeout(function() { 
          document.body.removeChild(tempDiv); 
        }, 2000);
      } else {
        console.log('Erro no envio:', r ? r.erro : 'Resposta inválida');
        showToast('Erro: ' + (r && r.erro ? r.erro : 'falha'), 'error');
      }
    })
    .withFailureHandler(function(e) {
      txt.disabled = false;
      btn.disabled = false;
      console.error('Erro ao enviar mensagem:', e);
      showToast('Erro: ' + e.message, 'error');
    })
    .enviarMensagemChat({
      id: id, 
      de: 'TI', 
      nome: tec, 
      msg: msg, 
      token: _tokenSessao
    });
}
